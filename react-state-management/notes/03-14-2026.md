function Component() {
  const [coffes, setCoffes] = useState([
    { id: 0, name: "Coffe 1" },
    { id: 1, name: "Coffe 2" },
    { id: 2, name: "Coffe 3" },
    { id: 3, name: "Coffe 4" },
  ]);

  const [selectedCoffe, setSelectedCoffe] = useState(null);

  function handleCoffeChange(coffe) {
    setSelectedCoffe(coffe);
  }
}

//right
function Component() {
  const [coffes, setCoffes] = useState([
    { id: 0, name: "Coffe 1" },
    { id: 1, name: "Coffe 2" },
    { id: 2, name: "Coffe 3" },
    { id: 3, name: "Coffe 4" },
  ]);

  const [selectedCoffeId, setSelectedCoffeId] = useState(null);

  function handleCoffeChange(coffe) {
    setSelectedCoffeId(coffe.id);
  }
}

//right
